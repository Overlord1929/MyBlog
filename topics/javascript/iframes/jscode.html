function goPlay(){
   if(myVideo.paused)
      myVideo.play();
}

function goPause(){
   if(myVideo.play)
      myVideo.pause();
}

function makeBigger(){
   if(myVideo.paused)
      myVideo.width=720;
}

function makeNormal(){
   if(myVideo.paused)
      myVideo.width=600;
}

function makeSmaller(){
   if(myVideo.paused)
      myVideo.width=480;
}
